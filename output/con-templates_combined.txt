        $(document).ready(function () {
            // Function to determine bar color based on average days and target
            function getBarColor(averageDays, targetDays) {
                if (averageDays < 2) {
                    return 'rgba(244, 67, 54, 0.6)'; // Red
                } else if (averageDays >= 2 && averageDays < targetDays) {
                    return 'rgba(255, 193, 7, 0.6)'; // Yellow
                } else {
                    return 'rgba(76, 175, 80, 0.6)'; // Green
                }
            }

            // Function to parse float safely
            function parseFloatSafe(value) {
                var parsed = parseFloat(value);
                return isNaN(parsed) ? 0 : parsed;
            }

            var ctx = document.getElementById('attendanceChart').getContext('2d');
            var averageDays = parseFloatSafe({{ printf "%.2f" .AverageDays }});
            var targetDays = parseFloatSafe({{ if .Preferences.TargetDays }}{{ printf "%.2f" .Preferences.TargetDays }}{{ else }}2.5{{ end }});
        

             var attendanceChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['In Office'],
                    datasets: [{
                        label: 'Days per Week',
                    data: [averageDays],
                    backgroundColor: [getBarColor(averageDays, targetDays)],
                    borderColor: [getBarColor(averageDays, targetDays)],
                    borderWidth: 1,
                    maxBarThickness: 50
        }]
    },
        options: {
            indexAxis: 'y',
                scales: {
                x: {
                    beginAtZero: true,
                        max: 5,
                            ticks: {
                        stepSize: 1
                    },
                    grid: {
                        display: false
                    }
                },
                y: {
                    display: false
                }
            },
            plugins: {
                legend: {
                    display: false
                },
                tooltip: {
                    callbacks: {
                        label: function(context) {
                            return context.parsed.x + ' Days';
                        }
                    }
                }
            },
            responsive: true,
                maintainAspectRatio: false // This line ensures the chart uses the container's height
        }
});
        // Handle click on attendance toggle
        $('.toggle-attendance').on('click', function () {
            var span = $(this);
            var date = span.data('date');
            var currentStatus = span.data('status'); // 'in' or 'remote'

            // Send AJAX POST request to toggle attendance
            $.ajax({
                url: '/toggle-attendance',
                method: 'POST',
                contentType: 'application/json',
                data: JSON.stringify({ date: date }),
                success: function (response) {
                    if (response.success) {
                        console.log("Success:", response);
                        // Update the span's class and text
                        if (response.newStatus === 'in') {
                            span.removeClass('event-remote').addClass('event-in-office');
                            span.html('<i class="fa-solid fa-building"></i> In Office');
                            span.data('status', 'in');
                        } else {
                            span.removeClass('event-in-office').addClass('event-remote');
                            span.html('<i class="fa-solid fa-home"></i> Remote');
                            span.data('status', 'remote');
                        }

                        // Update counts and averages
                        var averageDays = parseFloatSafe(response.averageDays).toFixed(2);
                        var targetDays = parseFloatSafe(response.targetDays).toFixed(2);
                        var inOfficeCount = response.inOfficeCount;
                        var totalDays = response.totalDays;
                        var averagePercent = parseFloatSafe(response.average).toFixed(2);

                        $('#average-days').text('In-Office Average Days for Q4 (Oct 1 - Dec 31): ' + averageDays + ' Days per Week');
                        $('#counts').text('In-Office Days: ' + inOfficeCount + ' / Total Days: ' + totalDays + ' / Ave Days: ' + averageDays);
                        $('#progress-fill').css('width', averagePercent + '%');
                        $('#progress-percent').text(averagePercent + '% In Office');

                        // Update Chart.js data
                        attendanceChart.data.datasets[0].data = [averageDays];
                        attendanceChart.data.datasets[0].backgroundColor = [getBarColor(averageDays, targetDays)];
                        attendanceChart.data.datasets[0].borderColor = [getBarColor(averageDays, targetDays)];
                        attendanceChart.update();

                        // Optionally, show a success message using Toastr
                        toastr.success('Attendance status updated successfully.');
                    } else {
                        // Optionally, show an error message using Toastr
                        toastr.error('Failed to update attendance status: ' + response.message);
                    }
                },
                error: function () {
                    toastr.error('An error occurred while updating attendance status.');
                }
            });
        });
        });

                            <button class="delete-button" data-id="{{.ID}}" title="Delete Vacation">
                                <i class="fa-solid fa-xmark"></i>

    <script>
        $(document).ready(function () {
            // Handle toggle for filtering attendance events
            $('#filterAttendance').on('change', function () {
                var isChecked = $(this).is(':checked');

                if (isChecked) {
                    // Show all events
                    $('.event-item').show();
                } else {
                    // Hide attendance events
                    $('.event-item').each(function () {
                        var eventType = $(this).data('type');
                        if (eventType === 'attendance') {
                            $(this).hide();
                        } else {
                            $(this).show();
                        }
                    });
                }
            });

            // Optional: Preserve toggle state using localStorage
            var toggleState = localStorage.getItem('filterAttendance');
            if (toggleState !== null) {
                $('#filterAttendance').prop('checked', toggleState === 'true').trigger('change');
            }

            $('#filterAttendance').on('change', function () {
                var isChecked = $(this).is(':checked');
                localStorage.setItem('filterAttendance', isChecked);
            });

            // Handle delete button click
            $('.delete-button').on('click', function () {
                var button = $(this);
                var eventId = button.data('id');

                if (confirm('Are you sure you want to delete this vacation event?')) {
                    // Send AJAX request to delete the event
                    $.ajax({
                        url: '/events/delete/' + eventId,
                        method: 'DELETE',
                        success: function (response) {
                            if (response.success) {
                                // Remove the event item from the list
                                button.closest('.event-item').fadeOut(300, function() {
                                    $(this).remove();
                                });
                                // Optionally, display a success message
                                toastr.success('Vacation event deleted successfully.');
                            } else {
                                // Optionally, display an error message
                                toastr.error('Failed to delete the vacation event.');
                            }
                        },
                        error: function () {
                            toastr.error('An error occurred while deleting the event.');
                        }
                    });
                }
            });
        });
    </script>
